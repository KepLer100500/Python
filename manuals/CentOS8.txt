### Создание пользователя

adduser kepler           # Создать нового пользователя в системе
passwd kepler            # Задать пароль
usermod -aG wheel kepler # Пользователям, принадлежащим к группе wheel, разрешается использовать команду sudo

### Локали

localectl                            # Отобразить активную locale
dnf install glibc-langpack-ru        # Устанавливаем русскую локаль
